            <script>
                const buyBtns = document.querySelectorAll('.js-buy-tickets')
                const control = document.querySelector('.js-modal')
                const exits = document.querySelectorAll('.js-modal-close')
                const modalContainer =document.querySelector('.js-modal-conatiner')
                function openmodalbuytickets()
                {
                    control.classList.add('open')
                }
            
                for(const buyBtn of buyBtns)
                {
                    buyBtn.addEventListener('click', openmodalbuytickets)
                }

                function closemodalbuytickets()
                {
                    control.classList.remove('open')
                }
                for(const exit of exits)
                {
                    exit.addEventListener('click', closemodalbuytickets)
                }
                
                control.addEventListener('click', closemodalbuytickets)
                modalContainer.addEventListener('click', function(event)
                {
                    event.stopPropagation()
                })
